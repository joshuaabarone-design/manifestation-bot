// Fetch agent config
const response = await fetch('https://your-url/api/agents/22/public');
const agent = await response.json();
// Build CALM-compliant system prompt
const systemPrompt = `
# IDENTITY
You are ${agent.name}. ${agent.characterBible.coreIdentity}
# ROLE
${agent.role}
# PERSONALITY (Big Five)
- Openness: ${agent.characterBible.personalityTraits.openness}/100
- Conscientiousness: ${agent.characterBible.personalityTraits.conscientiousness}/100
- Communication: ${agent.characterBible.communicationStyle.tone}, ${agent.characterBible.communicationStyle.formality}
# CONSTITUTIONAL RULES (MUST FOLLOW)
${agent.constitutionalRules.behavioralRules.map(r => `- [${r.priority}] ${r.rule}`).join('\n')}
# SAFETY BOUNDARIES (NEVER VIOLATE)
${agent.constitutionalRules.safetyBoundaries.map(b => `- ${b}`).join('\n')}
# PROHIBITED BEHAVIORS
${agent.constitutionalRules.prohibitedBehaviors.map(p => `- ${p}`).join('\n')}
# CAPABILITIES
${agent.capabilities.join('\n')}
`;
